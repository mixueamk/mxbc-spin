    <div class="logo">🍦 蜜雪冰城 · 波动巴西店</div>
    <div class="container">
        <div class="wheel-container">
            <div class="pointer">▼</div>
            <canvas id="wheelCanvas" width="560" height="560"></canvas>
        </div>
        <input type="tel" id="phoneInput" placeholder="请输入新加坡手机号 (9xxxxxxx)" />
        <button id="spinBtn" onclick="spin()">🎡 开始抽奖</button>
        <div class="message" id="resultMsg"></div>
        <div class="note">每天每号限抽1次，奖品到店出示群内中奖消息兑换</div>

    <script>
        // ========== 配置区 ==========
        const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbzmgWRNxMWb0DNwgkgAkgfuRGWyIy2Io__vXPxzUJeZFemur2lak5slL8D9LQdgduhoPQ/exec';
        const PRIZES = [
            { name: '整杯免单', color: '#FFD700', weight: 2 },
            { name: '免费加三份小料', color: '#00CED1', weight: 8 },
            { name: '芋圆一份免费', color: '#DA70D6', weight: 10 },
            { name: '蜜雪钥匙扣', color: '#FF8C00', weight: 15 },
            { name: '谢谢参与', color: '#B0C4DE', weight: 35 }
        ];
        const WEIGHTED_PRIZES = [];
        PRIZES.forEach(p => {
            for (let i = 0; i < p.weight; i++) WEIGHTED_PRIZES.push(p.name);
        });
        // ===========================

        const canvas = document.getElementById('wheelCanvas');
        const ctx = canvas.getContext('2d');
        const spinBtn = document.getElementById('spinBtn');
        const phoneInput = document.getElementById('phoneInput');
        const resultMsg = document.getElementById('resultMsg');
        let spinning = false;
        let currentPrize = null;

        function drawWheel(rotation = 0) {
            const center = 280;
            const radius = 260;
            const angleStep = (2 * Math.PI) / PRIZES.length;
            ctx.clearRect(0, 0, 560, 560);
            for (let i = 0; i < PRIZES.length; i++) {
                const startAngle = i * angleStep + rotation;
                const endAngle = startAngle + angleStep;
                ctx.beginPath();
                ctx.moveTo(center, center);
                ctx.arc(center, center, radius, startAngle, endAngle);
                ctx.closePath();
                ctx.fillStyle = PRIZES[i].color;
                ctx.fill();
                ctx.strokeStyle = 'white';
                ctx.lineWidth = 3;
                ctx.stroke();
                ctx.save();
                ctx.translate(center, center);
                ctx.rotate(startAngle + angleStep / 2);
                ctx.textAlign = "center";
                ctx.fillStyle = "#333";
                ctx.font = "bold 18px Arial";
                ctx.fillText(PRIZES[i].name, radius - 70, 8);
                ctx.restore();
            }
            ctx.beginPath();
            ctx.arc(center, center, 40, 0, 2*Math.PI);
            ctx.fillStyle = '#ff4d4d';
            ctx.fill();
            ctx.fillStyle = 'white';
            ctx.font = "bold 20px Arial";
            ctx.fillText("雪王", center, center+8);
        }

        function getRandomPrize() {
            const rand = Math.floor(Math.random() * WEIGHTED_PRIZES.length);
            return WEIGHTED_PRIZES[rand];
        }

        function isValidSgPhone(phone) {
            return /^[89]\d{7}$/.test(phone);
        }

        function hasSpunToday(phone) {
            const today = new Date().toDateString();
            const record = JSON.parse(localStorage.getItem('mxbc_spin')) || {};
            if (record[phone] && record[phone].date === today) {
                return true;
            }
            return false;
        }

        function saveSpinRecord(phone, prize) {
            const today = new Date().toDateString();
            const record = JSON.parse(localStorage.getItem('mxbc_spin')) || {};
            record[phone] = { date: today, prize: prize };
            localStorage.setItem('mxbc_spin', JSON.stringify(record));
        }

        async function logToSheet(phone, prize) {
            try {
                await fetch(SCRIPT_URL, {
                    method: 'POST',
                    mode: 'no-cors',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ phone: phone, prize: prize })
                });
            } catch (e) {
                console.log('记录失败，但可重试', e);
            }
        }

        async function spin() {
            if (spinning) return;
            const phone = phoneInput.value.trim();
            if (!isValidSgPhone(phone)) {
                alert('请输入有效的新加坡手机号（8位，9/8开头）');
                return;
            }
            if (hasSpunToday(phone)) {
                alert('您今天已经抽过啦，明天再来吧！');
                return;
            }
            spinning = true;
            spinBtn.disabled = true;
            resultMsg.textContent = '';

            currentPrize = getRandomPrize();
            const prizeIndex = PRIZES.findIndex(p => p.name === currentPrize);
            const anglePerPrize = 360 / PRIZES.length;
            const targetAngle = (360 - (prizeIndex * anglePerPrize + anglePerPrize / 2) + 90) % 360;
            const spins = 5 + Math.floor(Math.random() * 5);
            const finalRotation = 360 * spins + targetAngle;

            let currentRotation = parseFloat(canvas.dataset.rotation) || 0;
            const startRotation = currentRotation;
            const duration = 4000;
            const startTime = Date.now();

            function animate() {
                const now = Date.now();
                const elapsed = now - startTime;
                const progress = Math.min(elapsed / duration, 1);
                const ease = 1 - Math.pow(1 - progress, 3);
                const rotation = startRotation + finalRotation * ease;
                drawWheel((rotation * Math.PI) / 180);
                canvas.dataset.rotation = rotation;

                if (progress < 1) {
                    requestAnimationFrame(animate);
                } else {
                    spinning = false;
                    spinBtn.disabled = false;
                    resultMsg.textContent = `🎉 恭喜获得：${currentPrize} ！请截图，到店出示兑换。`;
                    saveSpinRecord(phone, currentPrize);
                    logToSheet(phone, currentPrize);
                }
            }

            requestAnimationFrame(animate);
        }

        drawWheel(0);
        canvas.dataset.rotation = 0;
    </script>
